
%!TEX root = oblig1.tex
%"Variabel" som holder oppgavenummer
\def\oppgaveNr{0}
\newcommand{\oppgave}[1]{
% Oppdater oppgavenummer
  \def\oppgaveNr{#1}
% Flytt marger
  \begin{addmargin}{-1.5em}
% Skriv oppgave og oppgavenummer
    {\normalfont\Large\bfseries Oppgave #1}
% Flytt marger tilbake
  \end{addmargin}
}
\newcommand{\deloppgave}[1]{
% Lag litt plass over
  \vspace*{1em}
% Flytt marger
  \begin{addmargin}{-0.5em}
% Skriv oppgave og oppgavenummer og bokstav
    {\normalfont\Large\bfseries \oppgaveNr #1)}
% Flytt marger tilbake
  \end{addmargin}
% Lag litt plass under
  \vspace*{1em}
}
